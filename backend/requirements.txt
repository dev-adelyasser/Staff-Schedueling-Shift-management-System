# =============================================================================
#  backend/requirements.txt — Staff Scheduling System
#
#  Dependency Philosophy
#  ─────────────────────
#  • All versions are pinned to the MINOR level (major.minor) to allow patch
#    security fixes while blocking breaking API changes.
#  • Groups are ordered: runtime → database → auth → testing → dev tooling.
#  • Versions were resolved against Python 3.11 as of June 2024.
#    Run `pip-compile requirements.in` to regenerate a fully locked file.
#
#  Testing Pyramid dependency map
#  ──────────────────────────────
#  Unit (70 %)        → pytest, pytest-cov, faker
#  Integration (20 %) → pytest-asyncio, httpx, factory-boy
#  E2E (10 %)         → playwright, pytest-playwright
#
#  Padlock Constraint Support
#  ──────────────────────────
#  pydantic v2 validators enforce boundary/threshold rules at schema layer.
#  python-jose + passlib enforce auth constraints at service layer.
# =============================================================================

# ── Web Framework ─────────────────────────────────────────────────────────────
fastapi==0.111.0
uvicorn[standard]==0.29.0       # includes websockets, httptools, uvloop

# ── Database ──────────────────────────────────────────────────────────────────
sqlalchemy==2.0.30              # 2.x ORM (mapped_column, DeclarativeBase)
alembic==1.13.1                 # schema migrations (never use create_all in prod)
psycopg2-binary==2.9.9          # sync postgres driver (binary = no libpq build dep)

# ── Validation & Settings ─────────────────────────────────────────────────────
pydantic==2.7.1                 # v2 — fast Rust core, strict mode, model_validator
pydantic-settings==2.2.1        # BaseSettings for env-var config (replaces v1 pattern)

# ── Authentication ────────────────────────────────────────────────────────────
python-jose[cryptography]==3.3.0  # JWT encoding / decoding
bcrypt==4.1.3                     # password hashing (avoids passlib+bcrypt edge cases)
python-multipart==0.0.9           # required by FastAPI for form data (OAuth2 flows)

# =============================================================================
#  TESTING DEPENDENCIES
#  These are required in all environments (dev, CI) that run the test suite.
#  In a production Docker image that runs only uvicorn, you can omit this
#  block by using pip install --no-deps or a separate requirements-prod.txt.
# =============================================================================

# ── Unit Layer Utilities (70 % of pyramid) ───────────────────────────────────
faker==25.2.0                   # realistic fake data for boundary/edge-case tests

# ── E2E Layer (10 % of pyramid — Phase 4) ────────────────────────────────────
playwright==1.44.0              # browser automation (Chromium / Firefox / WebKit)
pytest-playwright==0.4.4        # pytest plugin: --headed, --browser, auto page fixture
